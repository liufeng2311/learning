##############################################################
###                      Shrio的引入                       ###
##############################################################
1.引入Shrio的jar包
		<dependency>
			<groupId>org.apache.shiro</groupId>
			<artifactId>shiro-spring</artifactId>
			<version>1.4.0</version>
		</dependency>
2.项目引入Shrio(Shrio不像spring security引入jar包后会自动开启)
	初始化SecurityManager
	@Bean("securityManager")
    public SecurityManager securityManager(ShrioRealm myRealm,HashedCredentialsMatcher matcher) {
        DefaultWebSecurityManager securityManager = new DefaultWebSecurityManager();
        securityManager.setRealm(myRealm);   //自定义过滤规则(通过自定义的Realm实现对用户登录的校验)
        return securityManager;
    }
	开启Shrio功能
	@Bean("shiroFilter")
    public ShiroFilterFactoryBean shirFilter(SecurityManager securityManager) {
        ShiroFilterFactoryBean shiroFilter = new ShiroFilterFactoryBean();
        shiroFilter.setSecurityManager(securityManager);

        Map<String, Filter> filters = new HashMap<>();
        filters.put("myRealm", new ShrioFilter());
        shiroFilter.setFilters(filters);

        Map<String, String> filterMap = new LinkedHashMap<>();
        filterMap.put("/shiro/**", "anon");
        filterMap.put("/**", "myRealm");
        shiroFilter.setFilterChainDefinitionMap(filterMap);
        return shiroFilter;
    }
